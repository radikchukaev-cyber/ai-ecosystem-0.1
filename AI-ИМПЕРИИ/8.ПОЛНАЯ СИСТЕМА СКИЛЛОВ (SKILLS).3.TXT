markdown# ⚡ СУПЕРСКИЛЛ: Генератор кода (Fast Code Generator)

## ЧТО ЭТО

Скрипт который берёт описание функции (текст) и генерирует готовый рабочий код.
Не шаблоны. Не примеры. **ГОТОВЫЙ КОД** который можно скопировать и запустить.

## КОГДА ИСПОЛЬЗОВАТЬ

- Нужно написать функцию быстро
- Есть описание что нужно, но нет времени писать
- Нужно генерировать много функций одного типа
- Нужен стартовый код для дальнейшей разработки

## СКРИПТ (READY TO RUN)

``````python
#!/usr/bin/env python3
"""
VULCAN SUPERPOWER #1: Fast Code Generator
Генерирует рабочий код по описанию на русском
"""

import anthropic
import sys

def generate_code(description: str, language: str = "python") -> str:
    """
    Генерирует код по описанию
    
    Args:
        description: Описание что нужно написать
        language: Язык программирования (python, javascript, etc)
    
    Returns:
        Готовый код
    """
    client = anthropic.Anthropic()
    
    prompt = f"""Напиши готовый рабочий код на {language} языке.

ТРЕБОВАНИЕ: {description}

ПРАВИЛА:
- Код должен быть ГОТОВ к использованию (не примеры!)
- Включи error handling
- Добавь docstring
- Включи пример использования в конце
- Без комментариев на английском (только русский)
- Максимум 200 строк

ФОРМАТ ОТВЕТА:
`````{language}
[КОД]
`````

Больше ничего не пиши, только код в блоке."""

    message = client.messages.create(
        model="claude-opus-4-20250805",
        max_tokens=2000,
        messages=[
            {"role": "user", "content": prompt}
        ]
    )
    
    return message.content[0].text

def main():
    if len(sys.argv) < 2:
        description = input("📝 Опиши функцию которую нужно написать: ")
        language = input("🔤 На каком языке? (python/javascript/rust): ").strip() or "python"
    else:
        description = " ".join(sys.argv[1:])
        language = "python"
    
    print(f"\n⚙️ Генерирую код для: {description}\n")
    
    code = generate_code(description, language)
    print(code)
    
    # Сохраняю результат
    with open("generated_code.txt", "w") as f:
        f.write(code)
    print(f"\n✅ Код сохранён в generated_code.txt")

if __name__ == "__main__":
    main()
``````

``````markdown
## ПРИМЕР

Input: Медленный поиск дубликатов O(n²)
```python
def find_duplicates(data):
    duplicates = []
    for i in range(len(data)):
        for j in range(i+1, len(data)):
            if data[i] == data[j]:
                duplicates.append(data[i])
    return duplicates
```

Output:
````
## 🚀 ОПТИМИЗАЦИЯ #1: Использовать Set вместо вложенных циклов
- Проблема: O(n²) = на 1000 элементов это 1M операций!
- Причина: Вложенные циклы
- Решение: Использовать Set для O(n) сложности

````python
def find_duplicates(data):
    seen = set()
    duplicates = set()
    for item in data:
        if item in seen:
            duplicates.add(item)
        seen.add(item)
    return list(duplicates)
````

- Результат: 100-1000x БЫСТРЕЕ!
``````

``````markdown
`````python
#!/usr/bin/env python3
"""
VULCAN SUPERPOWER #5: Code Formatter
Форматирует код по стандартам + русские комментарии
"""

import subprocess
import sys
from pathlib import Path

def format_code(file_path: str) -> None:
    """
    Форматирует Python файл
    """
    path = Path(file_path)
    
    if not path.exists():
        print(f"❌ Файл не найден: {file_path}")
        return
    
    # Использую black (лучший форматер)
    try:
        subprocess.run(["black", str(path)], check=True)
        print(f"✅ Black форматирование: ОК")
    except subprocess.CalledProcessError:
        print("⚠️ Black форматирование: ошибка")
    
    # Использую isort (сортировка импортов)
    try:
        subprocess.run(["isort", str(path)], check=True)
        print(f"✅ Сортировка импортов: ОК")
    except subprocess.CalledProcessError:
        print("⚠️ Сортировка импортов: ошибка")
    
    # Проверяю PEP8
    try:
        subprocess.run(["pylint", str(path)], check=True)
        print(f"✅ PEP8 проверка: ОК")
    except subprocess.CalledProcessError:
        print("⚠️ PEP8 проверка: нужны улучшения")
    
    print(f"\n✅ Форматирование завершено: {file_path}")

def main():
    file_path = sys.argv[1] if len(sys.argv) > 1 else input("📁 Файл для форматирования: ")
    format_code(file_path)

if __name__ == "__main__":
    main()
``````

``````markdown
`````python
#!/usr/bin/env python3
"""
VULCAN SUPERPOWER #6: Dependency Manager
Управляет зависимостями проекта
"""

import subprocess
import json
import sys

def create_requirements(output_file: str = "requirements.txt") -> None:
    """
    Создаёт requirements.txt из установленных пакетов
    """
    try:
        result = subprocess.run(
            ["pip", "freeze"],
            capture_output=True,
            text=True,
            check=True
        )
        
        with open(output_file, "w") as f:
            f.write(result.stdout)
        
        print(f"✅ Создан {output_file}")
    except Exception as e:
        print(f"❌ Ошибка: {e}")

def check_vulnerabilities() -> None:
    """
    Проверяет уязвимости в зависимостях
    """
    try:
        subprocess.run(
            ["pip", "install", "safety"],
            capture_output=True
        )
        
        result = subprocess.run(
            ["safety", "check"],
            capture_output=True,
            text=True
        )
        
        print(result.stdout)
    except Exception as e:
        print(f"❌ Ошибка при проверке: {e}")

def update_dependencies() -> None:
    """
    Обновляет все зависимости
    """
    try:
        result = subprocess.run(
            ["pip", "list", "--outdated", "--format=json"],
            capture_output=True,
            text=True,
            check=True
        )
        
        outdated = json.loads(result.stdout)
        
        if not outdated:
            print("✅ Все зависимости обновлены")
            return
        
        print(f"📦 Найдено {len(outdated)} устаревших пакетов:")
        for pkg in outdated[:5]:  # Первые 5
            print(f"  - {pkg['name']}: {pkg['version']} → {pkg['latest_version']}")
        
        print("\n🔄 Обновляю...")
        subprocess.run(["pip", "install", "--upgrade", "pip"], check=True)
        
    except Exception as e:
        print(f"❌ Ошибка: {e}")

def main():
    if len(sys.argv) < 2:
        print("""
🛠️ DEPENDENCY MANAGER

Команды:
1. python vulcan_dependency_manager.py create    - создать requirements.txt
2. python vulcan_dependency_manager.py check     - проверить уязвимости
3. python vulcan_dependency_manager.py update    - обновить зависимости
        """)
        return
    
    command = sys.argv[1]
    
    if command == "create":
        create_requirements()
    elif command == "check":
        check_vulnerabilities()
    elif command == "update":
        update_dependencies()
    else:
        print(f"❌ Неизвестная команда: {command}")

if __name__ == "__main__":
    main()
`````
``````

``````markdown
`````python
#!/usr/bin/env python3
"""
VULCAN SUPERPOWER #9: Security Checker
Проверяет код на уязвимости
"""

import subprocess
import sys
from pathlib import Path

def check_security(file_path: str) -> None:
    """
    Полная проверка безопасности кода
    """
    path = Path(file_path)
    
    print(f"🔍 Сканирую {file_path} на уязвимости...\n")
    
    # 1. Bandit - поиск уязвимостей в Python
    print("1️⃣ Проверка Bandit (уязвимости)...")
    try:
        result = subprocess.run(
            ["bandit", "-r", str(path), "-f", "json"],
            capture_output=True,
            text=True
        )
        
        if "ISSUES FOUND" in result.stdout or result.returncode != 0:
            print("⚠️ Найдены потенциальные уязвимости!")
            subprocess.run(["bandit", str(path)])
        else:
            print("✅ Банди проверка пройдена")
    except FileNotFoundError:
        print("⚠️ Bandit не установлен: pip install bandit")
    
    # 2. Safety - проверка зависимостей
    print("\n2️⃣ Проверка Safety (зависимости)...")
    try:
        subprocess.run(["safety", "check"], check=False)
    except FileNotFoundError:
        print("⚠️ Safety не установлен: pip install safety")
    
    # 3. Pylint - проверка качества
    print("\n3️⃣ Проверка Pylint (качество)...")
    try:
        result = subprocess.run(
            ["pylint", str(path), "--exit-zero"],
            capture_output=True,
            text=True
        )
        print(result.stdout[-500:])  # Последние 500 символов
    except FileNotFoundError:
        print("⚠️ Pylint не установлен: pip install pylint")
    
    print("\n✅ Проверка безопасности завершена")

def main():
    file_path = sys.argv[1] if len(sys.argv) > 1 else input("📁 Файл для проверки: ")
    check_security(file_path)

if __name__ == "__main__":
    main()
`````
``````

```python
#!/usr/bin/env python3
"""
VULCAN SUPERPOWER: Obsidian Sync Manager
Синхронизирует Obsidian ↔ GitHub автоматически
"""

import os
import json
import subprocess
from pathlib import Path
from datetime import datetime
from git import Repo
from git.exc import InvalidGitRepositoryError

class ObsidianSyncManager:
    """
    Менеджер синхронизации Obsidian хранилища с GitHub
    """
    
    def __init__(self, obsidian_vault_path: str, git_repo_path: str = None):
        """
        Инициализация
        
        Args:
            obsidian_vault_path: Путь до Obsidian хранилища (например ~/Obsidian/Antigravity)
            git_repo_path: Путь до git репозитория (если None = используем obsidian_vault_path)
        """
        self.vault_path = Path(obsidian_vault_path).expanduser()
        self.repo_path = Path(git_repo_path or obsidian_vault_path).expanduser()
        
        if not self.vault_path.exists():
            raise FileNotFoundError(f"Хранилище не найдено: {self.vault_path}")
        
        try:
            self.repo = Repo(self.repo_path)
        except InvalidGitRepositoryError:
            print(f"⚠️ {self.repo_path} не git репозиторий")
            print("Инициализирую git...")
            self.repo = Repo.init(self.repo_path)
            self._setup_github_remote()
    
    def _setup_github_remote(self):
        """
        Настраивает GitHub remote если его нет
        """
        try:
            self.repo.create_remote('origin', self._get_github_url())
            print("✅ GitHub remote добавлен")
        except:
            print("✅ GitHub remote уже существует")
    
    def _get_github_url(self) -> str:
        """
        Получает URL GitHub репозитория от пользователя
        """
        url = input("🔗 GitHub репозиторий URL (например https://github.com/user/repo.git): ")
        return url
    
    def pull_from_github(self) -> bool:
        """
        Вытягивает изменения из GitHub в Obsidian
        """
        try:
            print("📥 Вытягиваю из GitHub...")
            self.repo.remotes.origin.pull()
            print("✅ Синхронизация из GitHub: успешно")
            return True
        except Exception as e:
            print(f"⚠️ Ошибка при pull: {e}")
            return False
    
    def push_to_github(self, message: str = None) -> bool:
        """
        Пушит изменения в GitHub
        
        Args:
            message: Сообщение коммита (если None генерируется автоматически)
        """
        try:
            # Добавляю все изменения
            self.repo.git.add('--all')
            
            # Проверяю есть ли что коммитить
            if not self.repo.is_dirty():
                print("✅ Нет изменений для коммита")
                return True
            
            # Создаю сообщение коммита
            if not message:
                changed_files = len(self.repo.index.diff('HEAD'))
                timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
                message = f"🔄 Sync Obsidian [{timestamp}] - {changed_files} файлов"
            
            # Коммитю
            self.repo.index.commit(message)
            print(f"✅ Коммит создан: {message}")
            
            # Пушу
            self.repo.remotes.origin.push()
            print("📤 Пушу в GitHub: успешно")
            return True
            
        except Exception as e:
            print(f"⚠️ Ошибка при push: {e}")
            return False
    
    def full_sync(self, direction: str = "both") -> bool:
        """
        Полная синхронизация
        
        Args:
            direction: "pull" (из GitHub), "push" (в GitHub), "both" (оба)
        """
        print(f"\n🔄 Полная синхронизация ({direction})...\n")
        
        success = True
        
        if direction in ["pull", "both"]:
            if not self.pull_from_github():
                success = False
        
        if direction in ["push", "both"]:
            if not self.push_to_github():
                success = False
        
        if success:
            print("\n✅ Синхронизация завершена успешно!")
        else:
            print("\n⚠️ Синхронизация завершена с ошибками")
        
        return success
    
    def get_status(self) -> dict:
        """
        Возвращает статус синхронизации
        """
        status = {
            "vault_path": str(self.vault_path),
            "repo_path": str(self.repo_path),
            "git_status": "initialized" if self.repo else "not_initialized",
            "has_remote": bool(self.repo.remotes) if self.repo else False,
            "is_dirty": self.repo.is_dirty() if self.repo else None,
            "last_commit": self.repo.head.commit.hexsha[:7] if self.repo.head.is_valid() else None,
            "last_commit_message": self.repo.head.commit.message if self.repo.head.is_valid() else None,
        }
        return status
    
    def print_status(self):
        """
        Красиво выводит статус
        """
        status = self.get_status()
        print(f"""
📊 СТАТУС СИНХРОНИЗАЦИИ

📁 Хранилище: {status['vault_path']}
🔧 Репозиторий: {status['repo_path']}
🔗 Git: {status['git_status']}
☁️ GitHub Remote: {'Да' if status['has_remote'] else 'Нет'}
📝 Изменения: {'Есть' if status['is_dirty'] else 'Нет'}
📌 Последний коммит: {status['last_commit']}
💬 Сообщение: {status['last_commit_message'][:50] if status['last_commit_message'] else 'N/A'}
        """)

def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="Obsidian Sync Manager")
    parser.add_argument("command", choices=["init", "status", "pull", "push", "sync"], 
                       help="Команда для выполнения")
    parser.add_argument("--vault", default="~/Obsidian/Antigravity", 
                       help="Путь до Obsidian хранилища")
    parser.add_argument("--repo", default=None,
                       help="Путь до git репозитория")
    parser.add_argument("--message", default=None,
                       help="Сообщение коммита для push")
    
    args = parser.parse_args()
    
    try:
        sync_manager = ObsidianSyncManager(args.vault, args.repo)
        
        if args.command == "status":
            sync_manager.print_status()
        
        elif args.command == "pull":
            sync_manager.pull_from_github()
        
        elif args.command == "push":
            sync_manager.push_to_github(args.message)
        
        elif args.command == "sync":
            sync_manager.full_sync("both")
        
        elif args.command == "init":
            print("✅ Инициализирован")
            sync_manager.print_status()
    
    except Exception as e:
        print(f"❌ Ошибка: {e}")

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
MNEMOSYNE SUPERPOWER: Vault Manager
Управляет структурой Obsidian хранилища по методу Карпаты
"""

import os
import json
from pathlib import Path
from datetime import datetime
import yaml

class VaultManager:
    """
    Менеджер Obsidian хранилища
    Следует методу Карпаты для организации файлов
    """
    
    # Правильная структура по Карпатам
    CARPATH_STRUCTURE = {
        "wiki": {
            "entities": {
                "System": {
                    "Configs": {"Root_Configs": {}},
                    "Logs": {},
                    "Audit": {},
                    "Tests": {"fixtures": {}, "mocks": {}},
                    "Scripts": {},
                    "Tools": {"Skills": {"references": {}, "integrations": {}}}
                },
                "Agents": {
                    "RAMS": {"memory": {}, "RAMS_skills": {}},
                    "VULCAN": {"memory": {}, "Vulcan_skills": {"scripts": {}}},
                    "ARGUS": {"memory": {}, "Argus_skills": {}},
                    "MNEMOSYNE": {"memory": {}, "Mnemosyne_skills": {}},
                    "SOCRATES": {"memory": {}, "Socrates_skills": {}},
                    "CASPER": {"memory": {}, "Casper_skills": {}},
                },
                "Companies": {}
            },
            "concepts": {},
            "sources": {},
            "synthesis": {
                "History": {},
                "Failures": {},
                "Canvases": {},
                "Legacy_2_Wiki": {},
                "Archive": {"backups": {}, "deprecated_concepts": {}}
            }
        },
        "raw": {
            "assets": {"youtube_thumbnails": {}, "screenshots": {}, "documents": {}},
            "sources": {},
            "transcripts": {"voice_notes": {}, "podcast_clips": {}},
            "web-clipped": {}
        }
    }
    
    def __init__(self, vault_path: str):
        """
        Инициализация менеджера
        
        Args:
            vault_path: Путь до Obsidian хранилища
        """
        self.vault_path = Path(vault_path).expanduser()
        self.vault_path.mkdir(parents=True, exist_ok=True)
    
    def create_directory_structure(self) -> bool:
        """
        Создаёт полную структуру по Карпатам
        """
        try:
            self._create_dirs(self.vault_path, self.CARPATH_STRUCTURE)
            print("✅ Структура хранилища создана по Карпатам")
            return True
        except Exception as e:
            print(f"❌ Ошибка при создании структуры: {e}")
            return False
    
    def _create_dirs(self, parent_path: Path, structure: dict):
        """
        Рекурсивно создаёт директории
        """
        for name, subfolders in structure.items():
            path = parent_path / name
            path.mkdir(parents=True, exist_ok=True)
            
            if isinstance(subfolders, dict):
                self._create_dirs(path, subfolders)
    
    def create_file_with_frontmatter(self, 
                                     file_path: str, 
                                     title: str,
                                     category: str = "general",
                                     tags: list = None,
                                     content: str = "") -> bool:
        """
        Создаёт файл с правильным YAML frontmatter
        
        Args:
            file_path: Путь до файла (относительно vault)
            title: Название файла
            category: Категория (для организации)
            tags: Список тегов
            content: Содержимое файла
        """
        try:
            full_path = self.vault_path / file_path
            full_path.parent.mkdir(parents=True, exist_ok=True)
            
            # Создаю frontmatter
            frontmatter = {
                "title": title,
                "date": datetime.now().strftime("%Y-%m-%d"),
                "category": category,
                "tags": tags or [],
                "status": "active",
                "path": str(file_path.rsplit('/', 1)[0] + "/" if "/" in file_path else "")
            }
            
            # Пишу файл
            with open(full_path, 'w', encoding='utf-8') as f:
                f.write("---\n")
                yaml.dump(frontmatter, f, allow_unicode=True)
                f.write("---\n\n")
                f.write(content)
            
            print(f"✅ Файл создан: {file_path}")
            return True
        
        except Exception as e:
            print(f"❌ Ошибка при создании файла: {e}")
            return False
    
    def create_index_file(self, 
                         category_path: str,
                         category_name: str) -> bool:
        """
        Создаёт индексный файл для категории
        
        Args:
            category_path: Путь до категории
            category_name: Название категории
        """
        index_content = f"""# 📑 Индекс {category_name}

Отсюда начинается навигация по {category_name}.

## Подкатегории

[Добавь ссылки на подкатегории здесь]

## Файлы

[Добавь ссылки на файлы здесь]

---
Последний обновлено: {datetime.now().strftime("%Y-%m-%d %H:%M")}
"""
        
        return self.create_file_with_frontmatter(
            f"{category_path}/INDEX.md",
            f"📑 Индекс {category_name}",
            category="index",
            tags=["#index", f"#{category_name.lower()}"],
            content=index_content
        )
    
    def validate_structure(self) -> dict:
        """
        Проверяет структуру по Карпатам
        
        Returns:
            Отчёт о проверке
        """
        report = {
            "valid": True,
            "errors": [],
            "warnings": [],
            "missing_dirs": [],
            "extra_dirs": [],
            "files_without_frontmatter": []
        }
        
        # Проверяю что все нужные директории есть
        def check_dirs(current_path: Path, expected: dict, prefix: str = ""):
            for name, subdirs in expected.items():
                dir_path = current_path / name
                full_name = f"{prefix}{name}" if not prefix else f"{prefix}/{name}"
                
                if not dir_path.exists():
                    report["missing_dirs"].append(full_name)
                    report["valid"] = False
                else:
                    if isinstance(subdirs, dict):
                        check_dirs(dir_path, subdirs, full_name)
        
        check_dirs(self.vault_path, self.CARPATH_STRUCTURE)
        
        # Проверяю файлы на frontmatter
        for md_file in self.vault_path.rglob("*.md"):
            if md_file.name in ["README.md", ".gitignore"]:
                continue
            
            with open(md_file, 'r', encoding='utf-8') as f:
                content = f.read()
                if not content.startswith("---"):
                    report["files_without_frontmatter"].append(str(md_file.relative_to(self.vault_path)))
        
        return report
    
    def print_report(self, report: dict):
        """
        Красиво выводит отчёт проверки
        """
        print(f"""
📊 ОТЧЁТ ПРОВЕРКИ СТРУКТУРЫ

{'✅ Структура правильная!' if report['valid'] else '❌ Найдены проблемы'}

Missing directories: {len(report['missing_dirs'])}
{chr(10).join(f'  ❌ {d}' for d in report['missing_dirs'][:5])}

Files without frontmatter: {len(report['files_without_frontmatter'])}
{chr(10).join(f'  ⚠️ {f}' for f in report['files_without_frontmatter'][:5])}

Status: {'🟢 OK' if report['valid'] else '🔴 NEEDS FIX'}
        """)
    
    def create_agent_vault(self, agent_name: str, skills: list = None) -> bool:
        """
        Создаёт полное хранилище для агента
        
        Args:
            agent_name: Имя агента (RAMS, VULCAN, и т.д.)
            skills: Список скиллов агента
        """
        try:
            base_path = f"wiki/entities/Agents/{agent_name}"
            
            # Создаю основные файлы
            self.create_file_with_frontmatter(
                f"{base_path}/identity.md",
                f"🧠 Личность {agent_name}",
                category="agent_identity",
                tags=[f"#{agent_name.lower()}", "#identity"],
                content=f"# {agent_name}\n\nЛичность агента {agent_name}."
            )
            
            self.create_file_with_frontmatter(
                f"{base_path}/soul.md",
                f"💫 Душа {agent_name}",
                category="agent_soul",
                tags=[f"#{agent_name.lower()}", "#soul"],
                content=f"# Миссия {agent_name}\n\nДуша и миссия {agent_name}."
            )
            
            self.create_file_with_frontmatter(
                f"{base_path}/laws.md",
                f"⚖️ Законы {agent_name}",

                tags=[f"#{agent_name.lower()}", "#laws"],

            # Создаю папку памяти
            memory_path = Path(self.vault_path) / base_path / "memory"
            memory_path.mkdir(parents=True, exist_ok=True)
            
            self.create_file_with_frontmatter(

            skills_path = Path(self.vault_path) / base_path / f"{agent_name.lower()}_skills"

        except Exception as e:
            print(f"❌ Ошибка при создании хранилища {agent_name}: {e}")
            return False
    
    def list_all_files(self) -> list:
        """
        Выводит список всех файлов в хранилище
        """
        files = []
        for md_file in self.vault_path.rglob("*.md"):
            relative_path = str(md_file.relative_to(self.vault_path))
            files.append(relative_path)
        return sorted(files)

def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="Vault Manager for Obsidian")
    parser.add_argument("command", 
                       choices=["init", "validate", "create-file", "create-agent", "list"],
                       help="Команда")
    parser.add_argument("--vault", default="~/Obsidian/Antigravity",
                       help="Путь до хранилища")
    parser.add_argument("--path", help="Путь до файла")
    parser.add_argument("--title", help="Название файла")
    parser.add_argument("--agent", help="Имя агента")
    parser.add_argument("--category", default="general", help="Категория")
    
    args = parser.parse_args()
    
    manager = VaultManager(args.vault)
    
    if args.command == "init":
        print("🔨 Создаю структуру по Карпатам...")
        manager.create_directory_structure()
        print("✅ Инициализация завершена")
    
    elif args.command == "validate":
        print("🔍 Проверяю структуру...")
        report = manager.validate_structure()
        manager.print_report(report)
    
    elif args.command == "create-file":
        if not args.path or not args.title:
            print("❌ Нужны --path и --title")
            return
        manager.create_file_with_frontmatter(args.path, args.title, args.category)
    
    elif args.command == "create-agent":
        if not args.agent:
            print("❌ Нужен --agent")
            return
        manager.create_agent_vault(args.agent)
    
    elif args.command == "list":
        files = manager.list_all_files()
        print(f"\n📚 {len(files)} файлов в хранилище:\n")
        for f in files[:20]:
            print(f"  📄 {f}")
        if len(files) > 20:
            print(f"  ... и ещё {len(files) - 20} файлов")

if __name__ == "__main__":
    main()
```
