markdown# 🎵 СУПЕРСКИЛЛ: Парсер мультимедиа (Multimedia Parser)

## ЧТО ЭТО

Агенты могут работать с ЛЮБЫМИ файлами:
- 🎤 **Аудио**: WAV, MP3, M4A, OGG (transcription + анализ)
- 🎬 **Видео**: MP4, MKV, AVI (извлечение frames, аудио, текст из видео)
- 🖼️ **Фото**: JPG, PNG, WebP (OCR, описание, анализ)
- 📄 **Документы**: PDF, DOCX, TXT (полный парсинг)
- 📊 **Архивы**: ZIP, RAR (автоматическое распаковывание)

Агент получает **чистый текст** из всего этого. Больше никаких "не умею работать с файлом".

## КОГДА ИСПОЛЬЗОВАТЬ

- Пришла голосовая сообщение в Telegram → парсим в текст
- Пришло видео с YouTube → извлекаем audio + распознаём речь
- Пришла фото с текстом → OCR → получаем текст
- Пришёл PDF → парсим всё содержимое
- Нужно обработать папку файлов → парсим все одновременно

```python
#!/usr/bin/env python3
"""
ARGUS SUPERPOWER: Multimedia Parser
Парсит любые медиа файлы в текст/данные
"""

import os
import json
import subprocess
import shutil
from pathlib import Path
from typing import Union, Dict, List
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class MultimediaParser:
    """
    Парсит аудио, видео, фото, документы в текст/метаданные
    """
    
    def __init__(self):
        """Инициализация с проверкой зависимостей"""
        self.dependencies = {
            'audio': ['ffmpeg', 'ffprobe'],
            'video': ['ffmpeg', 'ffprobe'],
            'image': ['tesseract'],  # OCR
            'pdf': ['pdftotext']
        }
        self._check_dependencies()
    
    def _check_dependencies(self):
        """Проверяет установлены ли нужные программы"""
        missing = []
        
        for dep_type, programs in self.dependencies.items():
            for prog in programs:
                if shutil.which(prog) is None:
                    missing.append(f"{prog} ({dep_type})")
        
        if missing:
            print(f"⚠️ Отсутствуют программы:")
            for m in missing:
                print(f"  pip install {m}")
    
    # ==================== АУДИО ====================
    
    def parse_audio(self, file_path: str) -> Dict:
        """
        Парсит аудиофайл:
        - Распознавание речи (speech-to-text)
        - Метаданные (длина, битрейт, язык)
        - Анализ (громкость, пики)
        """
        try:
            import librosa
            import numpy as np
            from speech_recognition import Recognizer, AudioFile
            
            file_path = Path(file_path)
            
            logger.info(f"🎵 Парсим аудио: {file_path}")
            
            # 1. Метаданные
            y, sr = librosa.load(str(file_path))
            duration = librosa.get_duration(y=y, sr=sr)
            
            # 2. Распознавание речи
            recognizer = Recognizer()
            with AudioFile(str(file_path)) as source:
                audio = recognizer.record(source)
            
            try:
                text = recognizer.recognize_google(audio, language='ru-RU')
                logger.info(f"✅ Текст распознан ({len(text)} символов)")
            except:
                text = "[Речь не распознана]"
            
            # 3. Анализ
            rms = librosa.feature.rms(y=y)[0]
            loudness = np.mean(rms)
            
            result = {
                "type": "audio",
                "file": str(file_path),
                "duration_seconds": float(duration),
                "sample_rate": sr,
                "text": text,
                "loudness": float(loudness),
                "language": "ru-RU"
            }
            
            return result
        
        except Exception as e:
            logger.error(f"❌ Ошибка при парсинге аудио: {e}")
            return {"type": "audio", "error": str(e)}
    
    # ==================== ВИДЕО ====================
    
    def parse_video(self, file_path: str, extract_frames: int = 3) -> Dict:
        """
        Парсит видео:
        - Извлечение frames (скриншоты)
        - Извлечение аудио + распознавание
        - Метаданные
        """
        try:
            from moviepy.editor import VideoFileClip
            from PIL import Image
            import io
            
            file_path = Path(file_path)
            logger.info(f"🎬 Парсим видео: {file_path}")
            
            video = VideoFileClip(str(file_path))
            duration = video.duration
            fps = video.fps
            
            # 1. Извлекаю audio
            audio_text = ""
            if video.audio is not None:
                audio_array = video.audio.to_soundarray()
                # Распознаю речь из видео
                audio_text = self._recognize_from_array(audio_array, video.audio.fps)
            
            # 2. Извлекаю frames
            frames_data = []
            frame_indices = [
                int(i * duration / extract_frames)
                for i in range(extract_frames)
            ]
            
            for idx in frame_indices:
                frame = video.get_frame(idx)
                # Конвертирую в текст через OCR если нужно
                frame_description = f"Frame at {idx}s"
                frames_data.append({
                    "timestamp": float(idx),
                    "description": frame_description
                })
            
            video.close()
            
            result = {
                "type": "video",
                "file": str(file_path),
                "duration_seconds": float(duration),
                "fps": fps,
                "audio_transcript": audio_text,
                "frames": frames_data,
                "resolution": f"{video.w}x{video.h}"
            }
            
            logger.info(f"✅ Видео распарсено ({len(frames_data)} frames)")
            return result
        
        except Exception as e:
            logger.error(f"❌ Ошибка при парсинге видео: {e}")
            return {"type": "video", "error": str(e)}
    
    # ==================== ФОТО ====================
    
    def parse_image(self, file_path: str, ocr: bool = True) -> Dict:
        """
        Парсит изображение:
        - OCR (распознавание текста в картинке)
        - Описание (что на картинке)
        - Метаданные
        """
        try:
            from PIL import Image
            from PIL.ExifTags import TAGS
            import pytesseract
            import anthropic
            
            file_path = Path(file_path)
            logger.info(f"🖼️ Парсим фото: {file_path}")
            
            image = Image.open(file_path)
            
            # 1. OCR (текст в картинке)
            text_in_image = ""
            if ocr:
                try:
                    text_in_image = pytesseract.image_to_string(image, lang='rus+eng')
                except:
                    text_in_image = "[OCR не доступен]"
            
            # 2. Метаданные
            exif_data = {}
            try:
                exif = image._getexif()
                if exif:
                    for (idx, tag), value in TAGS.items():
                        if idx in exif:
                            exif_data[tag] = str(value)[:100]
            except:
                pass
            
            # 3. Описание (через Claude vision)
            client = anthropic.Anthropic()
            
            import base64
            with open(file_path, "rb") as f:
                image_data = base64.standard_b64encode(f.read()).decode("utf-8")
            
            message = client.messages.create(
                model="claude-opus-4-20250805",
                max_tokens=500,
                messages=[
                    {
                        "role": "user",
                        "content": [
                            {
                                "type": "image",
                                "source": {
                                    "type": "base64",
                                    "media_type": "image/jpeg",
                                    "data": image_data,
                                },
                            },
                            {
                                "type": "text",
                                "text": "Опиши что ты видишь на этой фотографии. Будь краток."
                            }
                        ],
                    }
                ],
            )
            
            description = message.content[0].text
            
            result = {
                "type": "image",
                "file": str(file_path),
                "size": image.size,
                "format": image.format,
                "description": description,
                "ocr_text": text_in_image,
                "exif": exif_data
            }
            
            logger.info(f"✅ Фото распарсено")
            return result
        
        except Exception as e:
            logger.error(f"❌ Ошибка при парсинге фото: {e}")
            return {"type": "image", "error": str(e)}
    
    # ==================== PDF ====================
    
    def parse_pdf(self, file_path: str) -> Dict:
        """
        Парсит PDF:
        - Извлечение текста
        - Метаданные
        - Структура (главы, параграфы)
        """
        try:
            import PyPDF2
            
            file_path = Path(file_path)
            logger.info(f"📄 Парсим PDF: {file_path}")
            
            with open(file_path, 'rb') as f:
                pdf_reader = PyPDF2.PdfReader(f)
                
                # Метаданные
                metadata = pdf_reader.metadata
                
                # Текст
                text = ""
                pages = []
                
                for page_num, page in enumerate(pdf_reader.pages):
                    page_text = page.extract_text()
                    text += page_text + "\n"
                    pages.append({
                        "page": page_num + 1,
                        "text": page_text[:500]  # Первые 500 символов
                    })
            
            result = {
                "type": "pdf",
                "file": str(file_path),
                "pages": len(pdf_reader.pages),
                "title": metadata.title if metadata else "Unknown",
                "author": metadata.author if metadata else "Unknown",
                "full_text": text[:5000],  # Первые 5000 символов
                "page_previews": pages
            }
            
            logger.info(f"✅ PDF распарсено ({len(pdf_reader.pages)} страниц)")
            return result
        
        except Exception as e:
            logger.error(f"❌ Ошибка при парсинге PDF: {e}")
            return {"type": "pdf", "error": str(e)}
    
    # ==================== ДОКУМЕНТЫ ====================
    
    def parse_document(self, file_path: str) -> Dict:
        """
        Парсит документы (DOCX, TXT и т.д.)
        """
        try:
            file_path = Path(file_path)
            
            if file_path.suffix == '.txt':
                with open(file_path, 'r', encoding='utf-8') as f:
                    text = f.read()
                
                return {
                    "type": "text",
                    "file": str(file_path),
                    "text": text,
                    "lines": len(text.split('\n'))
                }
            
            elif file_path.suffix == '.docx':
                from docx import Document
                
                doc = Document(file_path)
                text = "\n".join([para.text for para in doc.paragraphs])
                
                return {
                    "type": "docx",
                    "file": str(file_path),
                    "text": text,
                    "paragraphs": len(doc.paragraphs)
                }
            
            else:
                return {"error": f"Неподдерживаемый формат: {file_path.suffix}"}
        
        except Exception as e:
            logger.error(f"❌ Ошибка при парсинге документа: {e}")
            return {"type": "document", "error": str(e)}
    
    # ==================== ОСНОВНОЙ МЕТОД ====================
    
    def parse(self, file_path: str) -> Dict:
        """
        Умный парсинг - определяет тип и парсит
        """
        file_path = Path(file_path)
        
        if not file_path.exists():
            return {"error": f"Файл не найден: {file_path}"}
        
        suffix = file_path.suffix.lower()
        
        logger.info(f"📂 Парсю файл: {file_path.name}")
        
        # Аудио
        if suffix in ['.mp3', '.wav', '.m4a', '.ogg', '.flac']:
            return self.parse_audio(str(file_path))
        
        # Видео
        elif suffix in ['.mp4', '.mkv', '.avi', '.mov', '.webm']:
            return self.parse_video(str(file_path))
        
        # Фото
        elif suffix in ['.jpg', '.jpeg', '.png', '.webp', '.bmp', '.gif']:
            return self.parse_image(str(file_path))
        
        # PDF
        elif suffix == '.pdf':
            return self.parse_pdf(str(file_path))
        
        # Документы
        elif suffix in ['.txt', '.docx', '.doc']:
            return self.parse_document(str(file_path))
        
        else:
            return {"error": f"Неподдерживаемый тип: {suffix}"}
    
    def parse_batch(self, folder_path: str) -> List[Dict]:
        """
        Парсит все файлы в папке
        """
        folder = Path(folder_path)
        results = []
        
        for file_path in folder.rglob('*'):
            if file_path.is_file():
                result = self.parse(str(file_path))
                results.append(result)
        
        logger.info(f"✅ Обработано файлов: {len(results)}")
        return results

def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="Multimedia Parser")
    parser.add_argument("file", help="Файл или папка для парсинга")
    parser.add_argument("--batch", action="store_true", help="Парсить всю папку")
    
    args = parser.parse_args()
    
    parser_instance = MultimediaParser()
    
    if args.batch:
        results = parser_instance.parse_batch(args.file)
    else:
        results = parser_instance.parse(args.file)
    
    print(json.dumps(results, indent=2, ensure_ascii=False))

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
VULCAN SUPERPOWER: GitHub Skills Manager
Управляет скиллами из GitHub репозиториев
"""

import os
import json
import subprocess
from pathlib import Path
from typing import List, Dict, Optional
from git import Repo
import requests
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class GitHubSkillsManager:
    """
    Менеджер для скачивания и управления скиллами из GitHub
    """
    
    def __init__(self, vault_path: str = "~/Obsidian/Antigravity"):
        """
        Инициализация
        
        Args:
            vault_path: Путь до Obsidian хранилища
        """
        self.vault_path = Path(vault_path).expanduser()
        self.skills_base = self.vault_path / "wiki" / "entities" / "System" / "Tools" / "Skills"
        self.skills_base.mkdir(parents=True, exist_ok=True)
        
        self.registry_file = self.skills_base / "SKILLS_REGISTRY.json"
        self._init_registry()
    
    def _init_registry(self):
        """Инициализирует реестр скиллов если его нет"""
        if not self.registry_file.exists():
            registry = {
                "skills": {},
                "last_updated": None,
                "source_repos": [
                    "https://github.com/iKiGAiyIS/Antigravity-Skills",
                    "https://github.com/iKiGAiyIS/Claude-Agents"
                ]
            }
            self._save_registry(registry)
    
    def _load_registry(self) -> Dict:
        """Загружает реестр скиллов"""
        with open(self.registry_file, 'r') as f:
            return json.load(f)
    
    def _save_registry(self, registry: Dict):
        """Сохраняет реестр скиллов"""
        with open(self.registry_file, 'w') as f:
            json.dump(registry, f, indent=2, ensure_ascii=False)
    
    def search_skill(self, skill_name: str) -> List[Dict]:
        """
        Ищет скилл в GitHub репозиториях
        
        Args:
            skill_name: Имя скилла (например "json_parser", "text_compressor")
        
        Returns:
            Список найденных скиллов с информацией
        """
        logger.info(f"🔍 Ищу скилл: {skill_name}")
        
        registry = self._load_registry()
        found = []
        
        # Ищу в локальном реестре
        for skill, info in registry.get("skills", {}).items():
            if skill_name.lower() in skill.lower():
                found.append(info)
        
        if found:
            logger.info(f"✅ Найдено в реестре: {len(found)} скиллов")
            return found
        
        # Ищу в GitHub
        logger.info("🌐 Ищу в GitHub репозиториях...")
        
        for repo_url in registry.get("source_repos", []):
            try:
                # Ищу через GitHub API
                owner_repo = repo_url.split("github.com/")[1].split(".git")[0]
                owner, repo = owner_repo.split("/")
                
                search_url = f"https://api.github.com/search/code?q=repo:{owner}/{repo}+{skill_name}+language:python"
                
                response = requests.get(search_url, timeout=10)
                
                if response.status_code == 200:
                    results = response.json().get("items", [])
                    for item in results[:5]:  # Первые 5
                        found.append({
                            "name": skill_name,
                            "file": item.get("path"),
                            "repo": repo_url,
                            "url": item.get("html_url"),
                            "size": item.get("size"),
                            "last_modified": "Unknown"
                        })
            
            except Exception as e:
                logger.warning(f"⚠️ Ошибка при поиске в {repo_url}: {e}")
        
        if found:
            logger.info(f"✅ Найдено в GitHub: {len(found)} скиллов")
        else:
            logger.warning(f"❌ Скилл не найден: {skill_name}")
        
        return found
    
    def download_skill(self, skill_name: str, source: str = None) -> bool:
        """
        Скачивает скилл из GitHub
        
        Args:
            skill_name: Имя скилла
            source: Источник (GitHub URL или локальный путь)
        
        Returns:
            True если успешно
        """
        # Ищу скилл
        results = self.search_skill(skill_name)
        
        if not results:
            logger.error(f"❌ Скилл не найден: {skill_name}")
            return False
        
        target = results[0] if not source else next(
            (r for r in results if r.get('url') == source), results[0]
        )
        
        logger.info(f"📥 Скачиваю: {target['name']}")
        
        try:
            # Скачиваю файл
            response = requests.get(target['url'].replace('github.com', 'raw.githubusercontent.com').replace('/blob/', '/'), timeout=30)
            
            if response.status_code != 200:
                logger.error(f"❌ Ошибка при скачивании: {response.status_code}")
                return False
            
            # Сохраняю в папку скиллов
            skill_folder = self.skills_base / skill_name
            skill_folder.mkdir(parents=True, exist_ok=True)
            
            skill_file = skill_folder / f"{skill_name}.py"
            
            with open(skill_file, 'w') as f:
                f.write(response.text)
            
            # Логирую в реестр
            self._register_skill(skill_name, target, str(skill_file))
            
            logger.info(f"✅ Скилл скачан: {skill_file}")
            return True
        
        except Exception as e:
            logger.error(f"❌ Ошибка при скачивании: {e}")
            return False
    
    def _register_skill(self, skill_name: str, info: Dict, local_path: str):
        """Регистрирует скилл в локальном реестре"""
        registry = self._load_registry()
        
        registry['skills'][skill_name] = {
            "name": skill_name,
            "path": local_path,
            "source": info.get('repo'),
            "source_url": info.get('url'),
            "installed": True,
            "installed_date": str(Path(local_path).stat().st_mtime),
            "size": info.get('size'),
            "tags": self._extract_tags(info.get('file', ''))
        }
        
        self._save_registry(registry)
    
    def _extract_tags(self, file_path: str) -> List[str]:
        """Извлекает теги из пути файла"""
        parts = file_path.split('/')
        return [p.lower().replace('-', '_') for p in parts if p]
    
    def list_installed(self) -> List[Dict]:
        """Выводит список установленных скиллов"""
        registry = self._load_registry()
        installed = [s for s in registry.get('skills', {}).values() if s.get('installed')]
        
        logger.info(f"📦 Установлено скиллов: {len(installed)}")
        
        for skill in installed:
            print(f"  ✓ {skill['name']:<30} {skill['path']}")
        
        return installed
    
    def get_skill(self, skill_name: str) -> Optional[str]:
        """
        Возвращает путь до установленного скилла
        
        Args:
            skill_name: Имя скилла
        
        Returns:
            Путь до файла скилла или None
        """
        registry = self._load_registry()
        
        skill = registry.get('skills', {}).get(skill_name)
        
        if skill and skill.get('installed'):
            path = Path(skill['path'])
            if path.exists():
                logger.info(f"✅ Скилл найден: {path}")
                return str(path)
        
        logger.warning(f"❌ Скилл не найден или не установлен: {skill_name}")
        return None
    
    def use_skill(self, skill_name: str, *args, **kwargs) -> Optional[str]:
        """
        Использует скилл (импортирует и запускает)
        
        Args:
            skill_name: Имя скилла
            *args: Аргументы для скилла
            **kwargs: Ключевые аргументы
        
        Returns:
            Результат выполнения скилла
        """
        skill_path = self.get_skill(skill_name)
        
        if not skill_path:
            logger.error(f"❌ Не могу использовать скилл: {skill_name}")
            return None
        
        try:
            # Динамический импорт
            import importlib.util
            spec = importlib.util.spec_from_file_location(skill_name, skill_path)
            module = importlib.util.module_from_spec(spec)
            spec.loader.exec_module(module)
            
            # Вызываю main функцию если есть
            if hasattr(module, 'main'):
                logger.info(f"🚀 Запускаю скилл: {skill_name}")
                return module.main(*args, **kwargs)
            
            logger.info(f"✅ Скилл загружен: {skill_name}")
            return module
        
        except Exception as e:
            logger.error(f"❌ Ошибка при использовании скилла: {e}")
            return None
    
    def update_registry(self):
        """Обновляет реестр скиллов из GitHub"""
        logger.info("🔄 Обновляю реестр скиллов...")
        
        registry = self._load_registry()
        
        for repo_url in registry.get("source_repos", []):
            try:
                # Клонирую репо временно
                temp_dir = Path("/tmp/skills_temp")
                
                if temp_dir.exists():
                    import shutil
                    shutil.rmtree(temp_dir)
                
                Repo.clone_from(repo_url, str(temp_dir), depth=1)
                
                # Ищу скилл файлы
                for py_file in temp_dir.rglob("*.py"):
                    if "skill" in py_file.name.lower():
                        skill_name = py_file.stem
                        
                        registry['skills'][skill_name] = {
                            "name": skill_name,
                            "repo": repo_url,
                            "file": str(py_file.relative_to(temp_dir)),
                            "installed": False
                        }
                
                logger.info(f"✅ Обновлён реестр из {repo_url}")
            
            except Exception as e:
                logger.warning(f"⚠️ Ошибка при обновлении: {e}")
        
        from datetime import datetime
        registry['last_updated'] = str(datetime.now())
        self._save_registry(registry)

def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="GitHub Skills Manager")
    parser.add_argument("command", 
                       choices=["search", "download", "list", "use", "update"],
                       help="Команда")
    parser.add_argument("--skill", help="Имя скилла")

if __name__ == "__main__":
```

```python
#!/usr/bin/env python3
"""
VULCAN SUPERPOWER: Open Source Skills Loader
Загружает скиллы из HuggingFace, PyPI, GitHub, и других источников
"""

import os
import json
import subprocess
from pathlib import Path
from typing import Dict, List, Optional
import requests
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class OpenSourceSkillsLoader:
    """
    Загружает скиллы из open source источников
    """
    
    def __init__(self, vault_path: str = "~/Obsidian/Antigravity"):
        self.vault_path = Path(vault_path).expanduser()
        self.skills_base = self.vault_path / "wiki" / "entities" / "System" / "Tools" / "Skills"
        self.skills_base.mkdir(parents=True, exist_ok=True)
        self.registry = self.skills_base / "OPENSOURCE_REGISTRY.json"
        
        self._init_registry()
    
    def _init_registry(self):
        """Инициализирует реестр open source скиллов"""
        if not self.registry.exists():
            registry = {
                "huggingface": {},
                "pypi": {},
                "gists": {},
                "custom": {},
                "loaded": []
            }
            with open(self.registry, 'w') as f:
                json.dump(registry, f, indent=2)
    
    # ==================== HUGGINGFACE ====================
    
    def search_huggingface(self, query: str, task: str = None) -> List[Dict]:
        """
        Ищет модели на HuggingFace
        
        Args:
            query: Поисковый запрос
            task: Тип задачи (text-generation, image-classification и т.д.)
        
        Returns:
            Список найденных моделей
        """
        try:
            logger.info(f"🔍 Ищу на HuggingFace: {query}")
            
            url = "https://huggingface.co/api/models"
            params = {"search": query}
            
            if task:
                params["filter"] = task
            
            response = requests.get(url, params=params, timeout=10)
            
            if response.status_code == 200:
                models = response.json()
                results = []
                
                for model in models[:10]:  # Первые 10
                    results.append({
                        "id": model.get("id"),
                        "name": model.get("modelId"),
                        "downloads": model.get("downloads", 0),
                        "likes": model.get("likes", 0),
                        "task": model.get("task"),
                        "url": f"https://huggingface.co/{model.get('id')}"
                    })
                
                logger.info(f"✅ Найдено: {len(results)} моделей")
                return results
            
            else:
                logger.error(f"❌ Ошибка HuggingFace API: {response.status_code}")
                return []
        
        except Exception as e:
            logger.error(f"❌ Ошибка при поиске: {e}")
            return []
    
    def download_huggingface_model(self, model_id: str) -> bool:
        """
        Скачивает модель с HuggingFace
        
        Args:
            model_id: ID модели (например "gpt2", "bert-base-uncased")
        
        Returns:
            True если успешно
        """
        try:
            from huggingface_hub import hf_hub_download
            
            logger.info(f"📥 Скачиваю модель: {model_id}")
            
            model_folder = self.skills_base / model_id.split("/")[-1]
            model_folder.mkdir(parents=True, exist_ok=True)
            
            # Скачиваю файлы модели
            files = ["config.json", "pytorch_model.bin", "tokenizer.json"]
            
            for file_name in files:
                try:
                    path = hf_hub_download(
                        repo_id=model_id,
                        filename=file_name,
                        cache_dir=str(model_folder)
                    )
                    logger.info(f"✅ Скачан: {file_name}")
                except:
                    pass  # Файл может не существовать
            
            self._register_huggingface(model_id, str(model_folder))
            logger.info(f"✅ Модель установлена: {model_id}")
            return True
        
        except Exception as e:
            logger.error(f"❌ Ошибка при скачивании: {e}")
            return False
    
    def _register_huggingface(self, model_id: str, path: str):
        """Регистрирует модель в реестре"""
        with open(self.registry, 'r') as f:
            registry = json.load(f)
        
        registry['huggingface'][model_id] = {
            "path": path,
            "installed": True,
            "source": "huggingface"
        }
        
        with open(self.registry, 'w') as f:
            json.dump(registry, f, indent=2)
    
    # ==================== PYPI ====================
    
    def search_pypi(self, query: str) -> List[Dict]:
        """
        Ищет пакеты на PyPI
        
        Args:
            query: Название пакета
        
        Returns:
            Список найденных пакетов
        """
        try:
            logger.info(f"🔍 Ищу на PyPI: {query}")
            
            url = f"https://pypi.org/pypi/{query}/json"
            response = requests.get(url, timeout=10)
            
            if response.status_code == 200:
                data = response.json()
                
                result = {
                    "name": data['info']['name'],
                    "version": data['info']['version'],
                    "summary": data['info']['summary'],
                    "author": data['info']['author'],
                    "url": data['info']['project_url'],
                    "downloads": "Unknown"
                }
                
                logger.info(f"✅ Найден пакет: {result['name']}")
                return [result]
            
            else:
                logger.warning(f"⚠️ Пакет не найден: {query}")
                return []
        
        except Exception as e:
            logger.error(f"❌ Ошибка при поиске: {e}")
            return []
    
    def install_pypi_package(self, package_name: str) -> bool:
        """
        Устанавливает пакет с PyPI
        
        Args:
            package_name: Название пакета
        
        Returns:
            True если успешно
        """
        try:
            logger.info(f"📥 Устанавливаю пакет: {package_name}")
            
            result = subprocess.run(
                ["pip", "install", package_name],
                capture_output=True,
                text=True,
                check=True
            )
            
            self._register_pypi(package_name)
            logger.info(f"✅ Пакет установлен: {package_name}")
            return True
        
        except subprocess.CalledProcessError as e:
            logger.error(f"❌ Ошибка при установке: {e.stderr}")
            return False
    
    def _register_pypi(self, package_name: str):
        """Регистрирует пакет в реестре"""
        with open(self.registry, 'r') as f:
            registry = json.load(f)
        
        registry['pypi'][package_name] = {
            "installed": True,
            "source": "pypi"
        }
        
        with open(self.registry, 'w') as f:
            json.dump(registry, f, indent=2)
    
    # ==================== GITHUB GISTS ====================
    
    def search_gists(self, query: str) -> List[Dict]:
        """
        Ищет скрипты в GitHub Gists
        
        Args:
            query: Поисковый запрос
        
        Returns:
            Список найденных гистов
        """
        try:
            logger.info(f"🔍 Ищу на GitHub Gists: {query}")
            
            url = "https://api.github.com/gists/public"
            
            results = []
            
            # GitHub API ограничивает поиск в гистах
            # Поэтому ищу через web
            web_url = f"https://gist.github.com/search?q={query}"
            response = requests.get(web_url, timeout=10)
            
            if response.status_code == 200:
                logger.info(f"✅ Найдены гисты для: {query}")
                return []  # Нужен парсинг HTML для полного списка
            
            return []
        
        except Exception as e:
            logger.warning(f"⚠️ Ошибка при поиске: {e}")
            return []
    
    def download_gist(self, gist_url: str) -> bool:
        """
        Скачивает скрипт из GitHub Gist
        
        Args:
            gist_url: URL гиста
        
        Returns:
            True если успешно
        """
        try:
            logger.info(f"📥 Скачиваю гист: {gist_url}")
            
            # Преобразую URL в raw версию
            raw_url = gist_url.replace("gist.github.com", "gist.githubusercontent.com").replace("/", "/raw/")
            
            response = requests.get(raw_url, timeout=10)
            
            if response.status_code == 200:
                # Определяю тип скрипта и сохраняю
                gist_name = gist_url.split("/")[-1][:8]
                
                skill_folder = self.skills_base / f"gist_{gist_name}"
                skill_folder.mkdir(parents=True, exist_ok=True)
                
                skill_file = skill_folder / "script.py"
                
                with open(skill_file, 'w') as f:
                    f.write(response.text)
                
                self._register_gist(gist_name, str(skill_file))
                logger.info(f"✅ Гист скачан: {skill_file}")
                return True
            
            else:
                logger.error(f"❌ Ошибка при скачивании")
                return False
        
        except Exception as e:
            logger.error(f"❌ Ошибка: {e}")
            return False
    
    def _register_gist(self, gist_id: str, path: str):
        """Регистрирует гист в реестре"""
        with open(self.registry, 'r') as f:
            registry = json.load(f)
        
        registry['gists'][gist_id] = {
            "path": path,
            "installed": True,
            "source": "github_gist"
        }
        
        with open(self.registry, 'w') as f:
            json.dump(registry, f, indent=2)
    
    # ==================== СПИСОК ВСЕГО ====================
    
    def list_all_skills(self) -> Dict:
        """Выводит список всех установленных скиллов из open source"""
        with open(self.registry, 'r') as f:
            registry = json.load(f)
        
        print(f"""
📦 УСТАНОВЛЕННЫЕ OPEN SOURCE СКИЛЛЫ

🤗 HuggingFace модели: {len(registry['huggingface'])}
{chr(10).join(f"  ✓ {k}" for k in list(registry['huggingface'].keys())[:5])}

📦 PyPI пакеты: {len(registry['pypi'])}
{chr(10).join(f"  ✓ {k}" for k in list(registry['pypi'].keys())[:5])}

🔗 GitHub Gists: {len(registry['gists'])}
{chr(10).join(f"  ✓ {k}" for k in list(registry['gists'].keys())[:5])}
        """)
        
        return registry

def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="Open Source Skills Loader")
    parser.add_argument("command",
                       choices=["search-hf", "download-hf", "search-pypi", "install-pypi", "download-gist", "list"],
                       help="Команда")
    parser.add_argument("--query", help="Поисковый запрос")
    parser.add_argument("--model", help="ID модели")
    parser.add_argument("--package", help="Название пакета")
    parser.add_argument("--gist", help="URL гиста")
    parser.add_argument("--vault", default="~/Obsidian/Antigravity", help="Путь до хранилища")
    
    args = parser.parse_args()
    
    loader = OpenSourceSkillsLoader(args.vault)
    
    if args.command == "search-hf":
        if not args.query:

            return
        
        results = loader.search_huggingface(args.query)
        print(json.dumps(results, indent=2, ensure_ascii=False))
    
    elif args.command == "download-hf":
```

markdown# 🗜️ СУПЕРСКИЛЛ: Компрессор данных (Data Compressor)

## ЧТО ЭТО

Берёт большой текст/данные → сжимает в 5-10 раз → сохраняет смысл → экономит токены.

Не просто удаляет пробелы, а **умно сжимает** - убирает мусор, оставляет суть.

## СКРИПТ

```python
#!/usr/bin/env python3
"""
ARGUS SUPERPOWER: Data Compressor
Сжимает данные сохраняя смысл (×5-10)
"""

import anthropic
import zstandard as zstd
from pathlib import Path
import json

class DataCompressor:
    """Компрессор данных"""
    
    def __init__(self):
        self.client = anthropic.Anthropic()
    
    def compress_text(self, text: str, target_ratio: float = 0.2) -> dict:
        """
        Сжимает текст на 80% (20% от оригинала)
        
        Args:
            text: Текст для сжатия
            target_ratio: Целевой размер (0.2 = 20%)
        
        Returns:
            Сжатый текст + метрики
        """
        original_size = len(text)
        target_size = int(original_size * target_ratio)
        
        prompt = f"""Сожми этот текст на {int((1-target_ratio)*100)}% сохраняя смысл.

ОРИГИНАЛ ({original_size} символов):
{text}

ТРЕБОВАНИЯ:
1. Оставь только суть (факты, выводы)
2. Удали примеры, истории, лишние подробности
3. Сохрани все цифры и даты
4. Результат должен быть ~{target_size} символов
5. Будь кратким (max {target_size} символов)

РЕЗУЛЬТАТ (только сжатый текст):"""
        
        response = self.client.messages.create(
            model="claude-opus-4-20250805",
            max_tokens=target_size + 500,
            messages=[{"role": "user", "content": prompt}]
        )
        
        compressed = response.content[0].text.strip()
        
        return {
            "original_size": original_size,
            "compressed_size": len(compressed),
            "ratio": len(compressed) / original_size,
            "compressed_text": compressed
        }
    
    def compress_json(self, data: dict) -> dict:
        """Сжимает JSON структуру"""
        # Удаляю ненужные поля
        cleaned = self._clean_json(data)
        
        original_size = len(json.dumps(data))
        compressed_size = len(json.dumps(cleaned))
        
        return {
            "original_size": original_size,
            "compressed_size": compressed_size,
            "ratio": compressed_size / original_size,
            "compressed_data": cleaned
        }
    
    def _clean_json(self, obj):
        """Удаляет пустые значения из JSON"""
        if isinstance(obj, dict):
            return {k: self._clean_json(v) for k, v in obj.items() 
                   if v and v != "" and v != [] and v != {}}
        elif isinstance(obj, list):
            return [self._clean_json(item) for item in obj]
        return obj
    
    def compress_file(self, file_path: str) -> dict:
        """Сжимает файл"""
        path = Path(file_path)
        
        with open(path, 'r') as f:
            content = f.read()
        
        if path.suffix == '.json':
            data = json.loads(content)
            return self.compress_json(data)
        else:
            return self.compress_text(content)

def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="Data Compressor")
    parser.add_argument("file", help="Файл для сжатия")
    parser.add_argument("--ratio", type=float, default=0.2, help="Целевое соотношение (0.2 = 20%)")
    
    args = parser.parse_args()
```

## ЧТО ЭТО

```python
#!/usr/bin/env python3
"""
ARGUS SUPERPOWER: Format Converter
Конвертирует между форматами
"""

import json
import csv
import yaml
from pathlib import Path
from typing import Union, Dict, List

class FormatConverter:
    """Конвертер форматов"""
    
    @staticmethod
    def json_to_yaml(data: Union[str, dict]) -> str:
        """JSON → YAML"""
        if isinstance(data, str):
            data = json.loads(data)
        
        return yaml.dump(data, allow_unicode=True, default_flow_style=False)
    
    @staticmethod
    def yaml_to_json(yaml_str: str) -> str:
        """YAML → JSON"""
        data = yaml.safe_load(yaml_str)
        return json.dumps(data, indent=2, ensure_ascii=False)
    
    @staticmethod
    def json_to_csv(data: Union[str, List[Dict]]) -> str:
        """JSON (список словарей) → CSV"""
        if isinstance(data, str):
            data = json.loads(data)
        
        if not data or not isinstance(data[0], dict):
            return "Error: JSON должен быть списком словарей"
        
        output = []
        fieldnames = list(data[0].keys())
        
        output.append(','.join(fieldnames))
        
        for row in data:
            values = [str(row.get(f, '')) for f in fieldnames]
            output.append(','.join(values))
        
        return '\n'.join(output)
    
    @staticmethod
    def csv_to_json(csv_str: str) -> str:
        """CSV → JSON"""
        lines = csv_str.strip().split('\n')
        reader = csv.DictReader(lines)
        data = list(reader)
        return json.dumps(data, indent=2, ensure_ascii=False)
    
    @staticmethod
    def markdown_to_html(markdown_str: str) -> str:
        """Markdown → HTML (простой парсер)"""
        try:
            import markdown
            return markdown.markdown(markdown_str)
        except:
            # Простой парсер если нет markdown
            html = markdown_str
            html = html.replace('#', '', 1).replace('\n', '\n', 1)
            html = html.replace('**', '')
            return f"{html}"
    
    @staticmethod
    def convert_file(input_file: str, output_format: str) -> str:
        """Конвертирует файл"""
        input_path = Path(input_file)
        input_format = input_path.suffix.lower().strip('.')
        
        with open(input_path, 'r', encoding='utf-8') as f:
            content = f.read()
        
        # Определяю что конвертировать
        if input_format == 'json' and output_format == 'yaml':
            result = FormatConverter.json_to_yaml(content)
        elif input_format == 'yaml' and output_format == 'json':
            result = FormatConverter.yaml_to_json(content)
        elif input_format == 'json' and output_format == 'csv':
            result = FormatConverter.json_to_csv(content)
        elif input_format == 'csv' and output_format == 'json':
            result = FormatConverter.csv_to_json(content)
        elif input_format == 'md' and output_format == 'html':
            result = FormatConverter.markdown_to_html(content)
        else:
            result = f"Конверсия {input_format} → {output_format} не поддерживается"
        
        # Сохраняю результат
        output_path = input_path.with_suffix(f'.{output_format}')
        
        with open(output_path, 'w', encoding='utf-8') as f:
            f.write(result)
        
        return str(output_path)

def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="Format Converter")
    parser.add_argument("input_file", help="Входной файл")
    parser.add_argument("output_format", choices=['json', 'yaml', 'csv', 'html', 'md'],
                       help="Формат на выходе")
    
    args = parser.parse_args()
    
    output = FormatConverter.convert_file(args.input_file, args.output_format)
    print(f"✅ Конвертировано: {output}")

if __name__ == "__main__":
    main()
```

## ЧТО ЭТО

Находит **дубликаты** в данных:
- Одинаковые строки
- Похожие строки (с опечатками)

## СКРИПТ

```python
#!/usr/bin/env python3
"""
ARGUS SUPERPOWER: Deduplicator
Находит и удаляет дубликаты
"""

import hashlib
from difflib import SequenceMatcher
from typing import List, Dict
import json

class Deduplicator:
    """Дедупликатор"""
    
    @staticmethod
    def find_exact_duplicates(items: List[str]) -> Dict:
        """Находит точные дубликаты"""
        seen = {}
        duplicates = {}
        
        for item in items:
            hash_val = hashlib.md5(item.encode()).hexdigest()
            
            if hash_val in seen:
                if hash_val not in duplicates:
                    duplicates[hash_val] = [seen[hash_val]]
                duplicates[hash_val].append(item)
            else:
                seen[hash_val] = item
        
        return {
            "total": len(items),
            "unique": len(seen),
            "duplicates": duplicates,
            "duplicate_count": sum(len(v)-1 for v in duplicates.values())
        }
    
    @staticmethod
    def find_similar_duplicates(items: List[str], threshold: float = 0.8) -> List[tuple]:
        """Находит похожие дубликаты"""
        similar = []
        
        for i, item1 in enumerate(items):
            for item2 in items[i+1:]:
                ratio = SequenceMatcher(None, item1, item2).ratio()
                
                if ratio > threshold:
                    similar.append({
                        "item1": item1,
                        "item2": item2,
                        "similarity": ratio
                    })
        
        return similar
    
    @staticmethod
    def deduplicate_list(items: List[str], keep_first: bool = True) -> List[str]:
        """Удаляет дубликаты из списка"""
        seen = set()
        result = []
        
        for item in items:
            hash_val = hashlib.md5(item.encode()).hexdigest()
            
            if hash_val not in seen:
                seen.add(hash_val)
                result.append(item)
        
        return result

def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="Deduplicator")
    parser.add_argument("file", help="Файл для проверки")
    parser.add_argument("--remove", action="store_true", help="Удалить дубликаты")
    
    args = parser.parse_args()
    
    with open(args.file, 'r') as f:
        lines = f.readlines()
    
    dedup = Deduplicator()
    result = dedup.find_exact_duplicates(lines)
    
    print(json.dumps(result, indent=2))
    
    if args.remove:
        cleaned = dedup.deduplicate_list(lines)
        with open(args.file + '.dedup', 'w') as f:
            f.writelines(cleaned)
        print(f"✅ Удалено {len(lines) - len(cleaned)} дубликатов")

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
ARGUS SUPERPOWER: Anomaly Detector
Находит аномалии в данных
"""

import json
import statistics
import anthropic
from typing import List, Union

class AnomalyDetector:
    """Детектор аномалий"""
    
    def __init__(self):
        self.client = anthropic.Anthropic()
    
    def detect_numeric_anomalies(self, numbers: List[float], std_threshold: float = 2.0) -> dict:
        """Находит аномалии в числовых данных"""
        if len(numbers) < 3:
            return {"error": "Нужно минимум 3 значения"}
        
        mean = statistics.mean(numbers)
        stdev = statistics.stdev(numbers)
        
        anomalies = []
        
        for i, num in enumerate(numbers):
            z_score = abs((num - mean) / stdev) if stdev > 0 else 0
            
            if z_score > std_threshold:
                anomalies.append({
                    "index": i,
                    "value": num,
                    "z_score": z_score,
                    "deviation": f"{((num - mean) / mean * 100):.1f}%"
                })
        
        return {
            "mean": mean,
            "stdev": stdev,
            "anomalies": anomalies,
            "anomaly_count": len(anomalies)
        }
    
    def detect_text_anomalies(self, texts: List[str]) -> dict:
        """Находит аномалии в текстовых данных"""
        prompt = f"""Найди аномальные тексты в этом списке:

{json.dumps(texts[:20], ensure_ascii=False)}

АНОМАЛИИ - это:
- Текст который сильно отличается от остальных
- Текст неправильного формата
- Текст с ошибками когда остальные правильные
- Текст с другим смыслом

ОТВЕТЬ JSON:
{{
  "anomalies": [
    {{
      "index": 0,
      "text": "...",
      "reason": "почему это аномалия"
    }}
  ],
  "total_anomalies": 2
}}"""
        
        response = self.client.messages.create(
            model="claude-opus-4-20250805",
            max_tokens=1000,
            messages=[{"role": "user", "content": prompt}]
        )
        
        result_text = response.content[0].text
        
        try:
            json_start = result_text.find('{')
            json_end = result_text.rfind('}') + 1
            return json.loads(result_text[json_start:json_end])
        except:
            return {"error": "Parse error"}

def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="Anomaly Detector")
    parser.add_argument("file", help="Файл для анализа (JSON с числами)")
    
    args = parser.parse_args()
    
    with open(args.file, 'r') as f:
        data = json.load(f)
    
    detector = AnomalyDetector()
    
    if isinstance(data, list) and all(isinstance(x, (int, float)) for x in data):
        result = detector.detect_numeric_anomalies(data)
```

markdown# ✔️ СУПЕРСКИЛЛ: Верификатор источников (Source Verifier)

## ЧТО ЭТО

Проверяет **надёжность источников**:
- Источник известен?
- Авторитетен ли?
- Проверено ли информация?
- Есть ли конфликт интересов?

## СКРИПТ

```python
#!/usr/bin/env python3
"""
ARGUS SUPERPOWER: Source Verifier
Проверяет надёжность источников
"""

import json
import anthropic
import requests
from typing import Dict, List

class SourceVerifier:
    """Верификатор источников"""
    
    def __init__(self):
        self.client = anthropic.Anthropic()
        
        self.trusted_domains = {
            "wikipedia.org", "github.com", "stackoverflow.com",
            "papers.arxiv.org", "scholar.google.com",
            "python.org", "official documentation"
        }
    
    def verify_url(self, url: str) -> dict:
        """Проверяет URL источника"""
        try:
            response = requests.head(url, timeout=5, allow_redirects=True)
            
            is_accessible = response.status_code == 200
            
            return {
                "url": url,
                "accessible": is_accessible,
                "status_code": response.status_code,
                "last_modified": response.headers.get('Last-Modified', 'Unknown'),
                "content_type": response.headers.get('Content-Type', 'Unknown')
            }
        
        except Exception as e:
            return {
                "url": url,
                "accessible": False,
                "error": str(e)
            }
    
    def assess_source_reliability(self, source_info: Dict) -> dict:
        """Оценивает надёжность источника"""
        prompt = f"""Оцени надёжность этого источника:

{json.dumps(source_info, ensure_ascii=False)}

ОЦЕНИВАЙ:
1. Авторитетность
2. Наличие проверок
3. Потенциальные конфликты интересов
4. Возраст информации

ОТВЕТЬ JSON:
{{
  "reliability_score": 0-100,
  "credibility": "HIGH|MEDIUM|LOW",
  "trust_level": "VERIFIED|PROBABLY_RELIABLE|QUESTIONABLE|UNRELIABLE",
  "concerns": ["проблема 1"],
  "recommendation": "что делать"
}}"""
        
        response = self.client.messages.create(
            model="claude-opus-4-20250805",
            max_tokens=800,
            messages=[{"role": "user", "content": prompt}]
        )
        
        result_text = response.content[0].text
        
        try:
            json_start = result_text.find('{')
            json_end = result_text.rfind('}') + 1
            return json.loads(result_text[json_start:json_end])
        except:
            return {"error": "Parse error"}

def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="Source Verifier")
    parser.add_argument("url", help="URL источника")
    
    args = parser.parse_args()
    
    verifier = SourceVerifier()
    
    print("🔍 Проверка источника...")
    
    # Проверяю доступность
    url_info = verifier.verify_url(args.url)
    print(json.dumps(url_info, indent=2))
    
    # Оцениваю надёжность
    print("\n📊 Оценка надёжности...")
    reliability = verifier.assess_source_reliability(url_info)
    print(json.dumps(reliability, indent=2))

if __name__ == "__main__":
    main()
```

```python
#!/usr/bin/env python3
"""
ARGUS SUPERPOWER: Content Classifier
Классифицирует контент
"""

import json
import anthropic

class ContentClassifier:
    """Классификатор контента"""
    
    def __init__(self):
        self.client = anthropic.Anthropic()
    
    def classify(self, content: str, categories: list = None) -> dict:
        """
        Классифицирует контент
        """
        categories_text = "\n".join(categories) if categories else """
- Тип: статья, новость, гайд, исследование, отзыв, другое
- Тема: AI, Python, Web, Data, DevOps, другое
- Уровень: начинающие, средний, продвинутый
- Язык: русский, английский, другой
"""
        
        prompt = f"""Классифицируй этот контент:

{content[:1000]}

КАТЕГОРИИ:
{categories_text}

ОТВЕТЬ JSON:
{{
  "type": "...",
  "topic": "...",
  "level": "...",
  "language": "...",
  "keywords": ["ключевое слово 1", "ключевое слово 2"],
  "summary": "краткое описание"
}}"""
        
        response = self.client.messages.create(
            model="claude-opus-4-20250805",
            max_tokens=500,
            messages=[{"role": "user", "content": prompt}]
        )
        
        result_text = response.content[0].text
        
        try:
            json_start = result_text.find('{')
            json_end = result_text.rfind('}') + 1
            return json.loads(result_text[json_start:json_end])
        except:
            return {"error": "Parse error"}

def main():
    import argparse
    
    parser = argparse.ArgumentParser(description="Content Classifier")
    parser.add_argument("file", help="Файл для классификации")
    
    args = parser.parse_args()
    
    with open(args.file, 'r') as f:
        content = f.read()
    
    classifier = ContentClassifier()
    result = classifier.classify(content)
    
    print(json.dumps(result, indent=2, ensure_ascii=False))

if __name__ == "__main__":
    main()
```
